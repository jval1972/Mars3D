// WP_FIST
WEAPON WP_FIST
{
  Ammo AM_NOAMMO
  States
  {
    Up:
    {
      FISG A 1 A_Raise
      Loop
    }
    Down:
    {
      FISG A 1 A_Lower
      Loop
    }
    Ready:
    {
      FISG A 1 A_WeaponReady
      Loop
    }
    Attack:
    {
      HADE B 4
      HADE C 4 A_Punch
      HADE B 4
      HADE D 4
      HADE E 4 A_Punch
      HADE D 4
      HADE A 5 A_Refire
      Goto Ready
    }
  }
}

// WP_PISTOL
WEAPON WP_PISTOL
{
  Ammo AM_BULLET
  States
  {
    Up:
    {
      LITG A 1 RAISE
      Loop
    }
    Down:
    {
      LITG A 1 LOWER
      Loop
    }
    Ready:
    {
      LITG A 1 WEAPONREADY
      Loop
    }
    Attack:
    {
      LITG A 3
      LITF A 4 FIREPISTOL
      LITF B 3
      LITF A 4 REFIRE
      Goto Ready
    }
    Flash:
    {
      TNT1 A 7 LIGHT1
      TNT1 E 0 LIGHT0
      Stop
    }
  }
}
